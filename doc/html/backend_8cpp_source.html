188 void CBackendx86::EmitScope(CScope *scope)
189 {
190  assert(scope != NULL);
191 
192  string label;
193 
194  if (scope->GetParent() == NULL) label = "main";
195  else label = scope->GetName();
196 
197  // label
198  _out << _ind << "# scope " << scope->GetName() << endl
199  << label << ":" << endl;
200 
201  // Compute stack offsets.
202  size_t stackOffsetResult = 0;
203  stackOffsetResult = ComputeStackOffsets(scope->GetSymbolTable(), 8, -12); // Compute stack offset.
204  _out << endl;
205 
206  // Function Prologue
207  stringstream content; // stringstream object for content of argument for EmitInstruction.
208 
209  _out << _ind << "# prologue" << endl;
210 
211  EmitInstruction("pushl", "%ebp"); // Push %ebp to stack.
212 
213  EmitInstruction("movl", "%esp, %ebp"); // Set %esp to %ebp.
214 
215  EmitInstruction("pushl", "%ebx", "save callee saved registers"); // Save callee saved register %ebx.
216  EmitInstruction("pushl", "%esi"); // Save callee saved register %esi.
217  EmitInstruction("pushl", "%edi"); // Save callee saved register %edi.
218  content << "$" << stackOffsetResult << ", " << "%esp";
219  EmitInstruction("subl", content.str(), "make room for locals"); // Make stack for local variables.
220 
221  if (stackOffsetResult > 0)
222  {
223  _out << endl;
224  }
225 
226  if (stackOffsetResult < 20 && stackOffsetResult > 0) // When stack offset is rather small. Manually set them. This may take at most 5 lines of assembly code.
227  {
228  EmitInstruction("xorl", "%eax, %eax", "memset local stack area to 0"); // Make %eax as zero.
229  int stackOffsetWordCnt = stackOffsetResult/4; // Get how many word(32-bit) should be initialized.
230  while (stackOffsetWordCnt > 0) // Initialize stack for local variables.
231  {
232  content.str("");
233  content << "%eax" << ", " << (stackOffsetWordCnt-1)*4 << "(%esp)";
234  //_out << _ind << "movl " << "%eax" << ", " << (stackOffsetWordCnt-1)*4 << "(%esp)" << endl;
235  EmitInstruction("movl", content.str());
236  stackOffsetWordCnt--;
237  }
238  //_out << endl;
239  }
240  else if (stackOffsetResult >= 20) // When stack offset is rather big. Use cld, rep, stosl to initialize stack for local variables.
241  {
242  EmitInstruction("cld", "", "memset local stack area to 0"); // Clear direction flags.
243  EmitInstruction("xorl", "%eax, %eax"); // Make %eax as zero.
244  content.str("");
245  content << "$" << stackOffsetResult/4 << ", " << "%ecx";
246  //_out << _ind << "movl " << "$" << stackOffsetResult/4 << ", " << "%ecx" << endl; // Calculate how many word(32-bit) should be initialized.
247  EmitInstruction("movl", content.str()); // Calculate how many word(32-bit) should be initialized.
248  EmitInstruction("mov", "%esp, %edi"); // Set relative offset for stosl instruction.
249  EmitInstruction("rep", "stosl"); // Repeat stosl for %ecx times.
250  //_out << endl;
251  // For more information of rep and stosl instruction, visit below two links.
252  // rep instruction : https://docs.oracle.com/cd/E19455-01/806-3773/instructionset-64/index.html
253  // stosl instruction : https://docs.oracle.com/cd/E19455-01/806-3773/instructionset-60/index.html
254  }
255 
256  // Emit Local Data.
257  if (scope->GetParent() != NULL)
258  {
259  EmitLocalData(GetScope());
260  }
261  else
262  {
263  _out << endl;
264  }
265 
266  // Function Body
267  _out << _ind << "# function body" << endl;
268 
269  EmitCodeBlock(scope->GetCodeBlock()); // Emit instructions in code block of given scope.
270 
271  _out << endl;
272 
273  // Function Epilogue
274  if (scope->GetParent() != NULL)
275  {
276  _out << "l_" << label << "_exit" << ":" << endl; // Function epilogue label.
277  }
278  else
279  {
280  _out << "l_" << _m->GetName() << "_exit" << ":" << endl;
281  }
282 
283  _out << _ind << "# epilogue" << endl;
284 
285  content.str(""); // Initialize content stringstream object.
286  content << "$" << stackOffsetResult << ", " << "%esp";
287  EmitInstruction("addl", content.str(), "remove locals"); // Remove locals on stack.
288 
289  EmitInstruction("popl", "%edi"); // Restore callee saved register %ebx.
290  EmitInstruction("popl", "%esi"); // Restore callee saved register %esi.
291  EmitInstruction("popl", "%ebx"); // Restore callee saved register %edi.
292 
293  EmitInstruction("popl", "%ebp"); // Restore base register %ebp.
294 
295  EmitInstruction("ret"); // Return from this function.
296 
297  _out << endl;

456 void CBackendx86::EmitInstruction(CTacInstr *i)
457 {
458  assert(i != NULL);
459 
460  ostringstream cmt;
461  ostringstream content;
462  string mnm;
463  cmt << i;
464 
465  EOperation op = i->GetOperation();
466 
467  switch (op) {
468  // binary operators
469  // dst = src1 op src2
470  case opAdd:
471  case opSub:
472  //
473  // For addition/subtraction of dst <- src1 +(-) src2, we follow the rule:
474  // 1) Load operand src1 to eax.
475  // 2) Load operand src2 to ebx.
476  // 3) Add/Subtract value of ebx to eax. (Instruction : addl/subl %ebx, %eax)
477  // 4) Store value of eax to dst.
478  //
479 
480  Load(i->GetSrc(1), "%eax", cmt.str());
481 
482  Load(i->GetSrc(2), "%ebx", "");
483 
484  content << "%ebx";
485  content << ", ";
486  content << "%eax";
487  if (op == opAdd) { EmitInstruction("addl", content.str()); } else { EmitInstruction("subl", content.str()); }
488  content.str("");
489 
490  Store(i->GetDest(), 'a', "");
491  break;
492 
493  case opMul:
494  case opDiv:
495  //
496  // For multiplication/division of dst <- src1 *(/) src2, we follow the rule:
497  // 1) Load operand src1 to eax.
498  // 2) Load operand src2 to ebx.
499  // 3) If opDiv, sign-extend [EAX] to [EDX:EAX]. (Instruction : cdq)
500  // 4) Multiply/Divide value of eax by ebx. (Instruction : imull/idivl %ebx)
501  // 5) Store value of eax to dst.
502  //
503 
504  Load (i->GetSrc(1), "%eax", cmt.str());
505 
506  Load (i->GetSrc(2), "%ebx", "");
507 
508  if (op == opDiv) { EmitInstruction("cdq", ""); }
509 
510  if (op == opMul) { EmitInstruction("imull", "%ebx"); } else { EmitInstruction("idivl", "%ebx"); }
511 
512  Store(i->GetDest(), 'a', "");
513  break;
514 
515  // unary operators
516  // dst = op src1
517  case opNeg:
518  //
519  // For unary negative, positive of dst <- src1, we follow the rule:
520  // 1) Load operand src1 to eax.
521  // 2) If opNeg, negate value. (Instruction : negl %eax)
522  // 3) Store value of eax to dst.
523  //
524  Load(i->GetSrc(1), "%eax", cmt.str());
525 
526  if (op == opNeg) { EmitInstruction("negl", "%eax"); }
527 
528  Store(i->GetDest(), 'a');
529  break;
530 
531  // memory operations
532  // dst = src1
533  case opAssign:
534  //
535  // For assignment of dst <- src1, we follow the rule:
536  // 1) Get operand src1 to eax.
537  // 2) Move value of eax to dst.
538  //
539 
540  Load(i->GetSrc(1), "%eax", cmt.str());
541 
542  Store(i->GetDest(), 'a', "");
543  break;
544 
545  // pointer operations
546  // dst = &src1
547  // dst = *src1
548  case opAddress:
549  //
550  // For reference operation of dst = &src1, we follow the rule:
551  // 1) Get address of src1 to eax. (Instruction : leal src1, %eax)
552  // 2) Store the value of eax to dst. (Instruction : movl %eax, dst)
553  //
554  content << Operand(i->GetSrc(1));
555  content << ", ";
556  content << "%eax";
557  EmitInstruction("leal", content.str(), cmt.str());
558  content.str("");
559 
560  Store(i->GetDest(), 'a');
561  break;
562 
563  case opDeref:
564  // opDeref not generated for now
565  EmitInstruction("# opDeref", "not implemented", cmt.str());
566  break;
567 
568 
569  // unconditional branching
570  // goto dst
571  case opGoto:
572  //
573  // For goto of goto dst, we follow the rule:
574  // 1) Unconditionally branch to l_(scopeName)_(dst).
575  //
576  EmitInstruction("jmp", Label(dynamic_cast<CTacLabel*>(i->GetDest())), cmt.str());
577  break;
578 
579  // conditional branching
580  // if src1 relOp src2 then goto dst
581  case opEqual:
582  case opNotEqual:
583  case opBiggerEqual:
584  case opBiggerThan:
585  case opLessEqual:
586  case opLessThan:
587  //
588  // For conditional branching of if src1 relOp src2 then goto dst, we follow the rule:
589  // 1) Load src1 to eax.
590  // 2) Load src2 to ebx.
591  // 3) Compare ebx to eax. (Instruction : cmp %ebx, %eax)
592  // 4) j(relOp) dst
593  //
594 
595  Load(i->GetSrc(1), "%eax", cmt.str());
596 
597  Load(i->GetSrc(2), "%ebx");
598 
599  EmitInstruction("cmpl", "%ebx, %eax");
600 
601  if (op == opEqual) { EmitInstruction("je", Label(dynamic_cast<CTacLabel*>(i->GetDest()))); }
602  else if (op == opNotEqual) { EmitInstruction("jne", Label(dynamic_cast<CTacLabel*>(i->GetDest()))); }
603  else if (op == opBiggerEqual) { EmitInstruction("jge", Label(dynamic_cast<CTacLabel*>(i->GetDest()))); }
604  else if (op == opBiggerThan) { EmitInstruction("jg", Label(dynamic_cast<CTacLabel*>(i->GetDest()))); }
605  else if (op == opLessEqual) { EmitInstruction("jle", Label(dynamic_cast<CTacLabel*>(i->GetDest()))); }
606  else if (op == opLessThan) { EmitInstruction("jl", Label(dynamic_cast<CTacLabel*>(i->GetDest()))); }
607  else { EmitInstruction("###", "Inavlid instruction."); }
608  break;
609 
610  // function call-related operations
611  case opParam:
612  //
613  // For adding parameter(s) of param (idx) <- src1, we follow the rule:
614  // 1) Load src1 to eax.
615  // 2) push eax to stack.
616  //
617 
618  Load(i->GetSrc(1), "%eax", cmt.str());
619 
620  EmitInstruction("pushl", "%eax");
621  break;
622 
623  case opCall:
624  //
625  // For calling function/procedure of call (dst) <- src1, we follow the rule:
626  // 1) Call (src1). Result will be stored at eax.
627  // 2) Add 4 to esp. This will get esp to original position.
628  // 3) If function, Store the value of eax to dst.
629  //
630 
631  EmitInstruction("call", dynamic_cast<CTacName*>(i->GetSrc(1))->GetSymbol()->GetName(), cmt.str());
632 
633  if (dynamic_cast<const CSymProc*>(dynamic_cast<CTacName*>(i->GetSrc(1))->GetSymbol())->GetNParams() > 0) // When parameters exist, add them.
634  {
635  content << "$";
636  content << 4 * dynamic_cast<const CSymProc*>(dynamic_cast<CTacName*>(i->GetSrc(1))->GetSymbol())->GetNParams();
637  content << ", ";
638  content << "%esp";
639 
640  EmitInstruction("addl", content.str());
641  content.str("");
642  }
643 
644  if (i->GetDest() != NULL)
645  {
646  Store(i->GetDest(), 'a');
647  }
648  break;
649 
650  case opReturn:
651  //
652  // For return of return (src1), we follow the rule:
653  // 1) Load src1 to eax.
654  // 2) Jump to l_(scopeName)_exit.
655  //
656 
657  if (i->GetSrc(1) != NULL) // When return value exist, load that.
658  {
659  Load(i->GetSrc(1), "%eax", cmt.str());
660  }
661 
662  content << "l_";
663  content << GetScope()->GetName();
664  content << "_exit";
665 
666  EmitInstruction("jmp", content.str());
667  break;
668 
669  // special
670  case opLabel:
671  _out << Label(dynamic_cast<CTacLabel*>(i)) << ":" << endl;
672  break;
673 
674  case opNop:
675  EmitInstruction("nop", "", cmt.str());
676  break;
677 
678 
679  default:
680  EmitInstruction("# ???", "not implemented", cmt.str());
681  }
682 }

868  int param_ofs,int local_ofs)
869 {
870  assert(symtab != NULL);
871  vector<CSymbol*> slist = symtab->GetSymbols();
872 
873  size_t localSymCnt = 0;
874 
875  int currentParamOffset = param_ofs; // This variable keeps track of parameter offset.
876  int currentLocalOffset = local_ofs; // This variable keeps track of stack offset.
877  size_t occupiedStack = 0; // This variable keeps track of how many stack is used.
878 
879  // For local symbols.
880  for (; localSymCnt < slist.size(); localSymCnt++)
881  {
882  CSymbol* lsym = slist[localSymCnt]; // Symbol.
883 
884  if (lsym->GetSymbolType() == stLocal) // Local symbol.
885  {
886  if (lsym->GetDataType()->IsInt()) // Case when symbol is integer type.
887  {
888  // Check for alignment
889  if (occupiedStack%4 != 0)
890  {
891  int alignRemainder = occupiedStack%4;
892  occupiedStack += (4 - alignRemainder);
893  currentLocalOffset -= (4 - alignRemainder);
894  }
895 
896  currentLocalOffset -= 4; // Occupy 4 bytes.
897  occupiedStack += 4; // Occupy 4 bytes.
898  lsym->SetOffset(currentLocalOffset); // Set offset of this symbol.
899  lsym->SetBaseRegister("%ebp"); // Set base register of this symbol.
900  }
901  else if (lsym->GetDataType()->IsBoolean()) // Case when symbol is boolean type.
902  {
903  currentLocalOffset -= 1; // Occupy 1 byte.
904  occupiedStack += 1; // Occupy 1 byte.
905  lsym->SetOffset(currentLocalOffset); // Set offset of this symbol.
906  lsym->SetBaseRegister("%ebp"); // Set base register of this symbol.
907  }
908  else if (lsym->GetDataType()->IsChar()) // Case when symbol is char type.
909  {
910  currentLocalOffset -= 1; // Occupy 1 byte.
911  occupiedStack += 1; // Occupy 1 byte.
912  lsym->SetOffset(currentLocalOffset); // Set offset of this symbol.
913  lsym->SetBaseRegister("%ebp"); // Set base register of this symbol.
914  }
915  else if (lsym->GetDataType()->IsArray()) // Case when symbol is array type.
916  {
917  // Check for alignment
918  if ((occupiedStack + lsym->GetDataType()->GetSize())%4 != 0)
919  {
920  int alignRemainder = (occupiedStack + lsym->GetDataType()->GetSize())%4; // For array, we align with after-pushing stack.
921  occupiedStack += (4 - alignRemainder);
922  currentLocalOffset -= (4 - alignRemainder);
923  }
924  int dataSize = lsym->GetDataType()->GetSize(); // Get the data size for this array, including metadata.
925  currentLocalOffset -= dataSize; // Occupy (datasize) bytes.
926  occupiedStack += dataSize; // Occupy (datasize) bytes.
927  lsym->SetOffset(currentLocalOffset); // Set offset of this symbol.
928  lsym->SetBaseRegister("%ebp"); // Set base register of this symbol.
929  }
930  else if (lsym->GetDataType()->IsPointer()) // Case when symbol is pointer type.
931  {
932  // Check for alignment

934  {
935  int alignRemainder = occupiedStack%4;
936  occupiedStack += (4 - alignRemainder);
937  currentLocalOffset -= (4 - alignRemainder);
938  }
939  currentLocalOffset -= 4; // Occupy 4 bytes.
940  occupiedStack += 4; // Occupy 4 bytes.
941  lsym->SetOffset(currentLocalOffset); // Set offset of this symbol.
942  lsym->SetBaseRegister("%ebp"); // Set base register of this symbol.
943  }
944  }
945  }
946

948  for (localSymCnt = 0; localSymCnt < slist.size(); localSymCnt++)

951 
952  if (lsym->GetSymbolType() == stParam) // Parameter.
953  {
954  if (lsym->GetDataType()->IsInt()) // Case when symbol is integer type.
955  {
956  lsym->SetOffset(currentParamOffset + 4 * (dynamic_cast<CSymParam*>(lsym)->GetIndex() - 1)); // Set offset of this symbol.
957  lsym->SetBaseRegister("%ebp"); // Set base register of this symbol.
958  }

967  lsym->SetBaseRegister("%ebp"); // Set base register of this symbol.
968  }
969  else if (lsym->GetDataType()->IsPointer()) // Case when symbol is pointer type.
970  {
971  lsym->SetOffset(currentParamOffset + 4 * (dynamic_cast<CSymParam*>(lsym)->GetIndex() - 1)); // Set offset of this symbol.
972  lsym->SetBaseRegister("%ebp"); // Set base register of this symbol.
973  }
974  }
975  }

978  // Dump stack frame to assembly file.
979
